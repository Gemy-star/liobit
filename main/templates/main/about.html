مقدمة

صـناعـة الإعـلام تـبدأ بـبناء اسـتراتـيجية إعـلامـية واضـحة دد خــــطوات الــــعمل تــــنتج عــــنها خــــطة مــــحكمة تحــــ ّ بـوضـوح ومـوضـوعـية ، يـقود نـجاحـها عـلاقـات إعـلامـية متينة. نــؤمــن بــأن الــعمل الإعــلامــي هــو عــمل مــؤســساتــي بـامـتياز وأن مـمارسـته كـنظام شـامـل ومـتكامـل هـو مـا تـحتاجـه الشـركـات والـجهات وجـميع الـكيانـات مـن أجـل بناء هيكل إعلامي قوي.

صـحيفة مـتخصصة فـي الـشأن الاقـتصادي تـصدر بـالـلغتين الـعربـية والانـكليزيـة، حـملت خـلال عـقد مــن الــزمــن لــقب الــوســيلة الإعــلامــية الــرقــمية ً الأولى في المجال الاقتصادي وأصبحت مرجعا تـتناقـل أخـبارهـا وكـالات الأنـباء المحـلية والـعالـمية لــما تحــظى بــه تــقاريــرهــا وتحــليلاتــها وأخــبارهــا بموثوقية عالمية المستوى

خدمات

الاتصالات التسويقية

عد الإعـــلان أحـــد أهـــم الـــوســـائـــل الـــتي تســـتخدمـــها يـــ ّ الــجهات والشــركــات لــترويــج مــنتجاتــها ، ولــكن لا يــمكن تحـديـد نـوع الإعـلان ومـاهـيته دون خـبرة حـقيقية فـي التسويق ، خططه وآلياته ، استراتيجيته وأساليبه. نــــقدم خــــبراتــــنا عــــلى مــــدى عــــقود فــــي الخــــدمــــات مـن بـناء وتـطويـر الإسـتراتـيجيات ً الـتسويـقية بـأنـواعـها بـدءا إلــــــى إعــــــداد ورســــــم الخــــــطط مــــــع تحــــــليل الــــــسوق والاحـتياجـات والشـرائـح المسـتهدفـة وإضـافـة إلـى دراسـة المنتج وتحديد المنافسين.

تنظيم المعارض

سهــ ًلا فــي الــبدايــة ولــكن ً إقــامــة فــعالــية قــد يــبدو أمــرا تـحويـل أي فـعالـية إلـى حـدث نـاجـح هـو أمـر يـتطلب فـريـق عمل محترف. نــــقدم فــــي مــــال أبــــرع الــــكفاءات فــــي تــــنظيم وإدارة الأحــداث بــأنــواعــها(مــعارض- مــؤتــمرات - فــعالــيات - مـن إنـشاء ً ورشعـمل - احـتفالـيات) بـكل تـفاصـيلها بـدءا بتخــطيطه وتــنفيذه وإضــافــة إلــى ً فــكرة الحــدث مــرورا تسويق الرعايات والترويج الإعلاني الاحترافي.

مركز الأبحاث والدراسات

أسست شركة مال الإعلامية الدولية مركز دراسات وأبحاث لجمع البيانات وتوفير المعلومات وعمل القراءات ووضع الأساسات ليكون شريكاً فاعلاً ، ومصدراً موثوقاً يمكن الاعتماد عليه في المساهمة بتسريع الخطى للشركات نحو المستقبل بوتيرة ثابتة ومتقنة.

العلاقات العامة

تــقوم الــعلاقــات الــعامــة عــلى أســسرئيســية ، تــجتمع لــتك ّ نظام عمل مثمر. نـعمل مـع عـملائـنا عـلى بـناء بـرامـج اسـتراتـيجية الـعلاقـات الـعامـة مـنذ الـبدايـة ، صـياغـة الخـطط والسـياسـات ، وتـنفيذهـا بـما يـحقق مصالحهم. نــقدم جــميع خــدمــاتــنا بــما فــيها الــمراكــز الإعــلامــية الاحــترافــية وبــرامــج الــمسؤولــية الاجــتماعــية وخــطط الــتواصــل الــداخــلي ، إضـافـة إلـى قـسم نشـر مـتخصصيـبرع فـي إعـداد الـتقاريـر الـسنويـة للشركات والمجلات المختصة والنشرات الإعلامية والتحليلية.

الإنتاج التلفزيوني

تصوير مقطع فيديو أمر سهل جداً باستعمال هاتف ذكي ، إلا أن صناعة فيلم مميّز هو رؤية وموهبة مع مجموعة كبيرة من المعدات والكاميرات وفريق عمل مختص. نقدم جميع خدمات التصوير والإنتاج التلفزيوني لعملائنا كالأفلام الوثائقية والإعلانات التلفزيونية وإعداد وتنفيذ اللقاءات الحوارية والبرامج المتلفزة.
